function savePromptToPocket(text) { if (!text || !text.trim()) return; let pocket = loadPromptPocket(); // Hindari duplikat persis di urutan teratas if (pocket.length && pocket[0].text.trim() === text.trim()) return; const now = new Date(); const pad = n => n.toString().padStart(2,'0'); const timestamp = `${now.getFullYear()}-${pad(now.getMonth()+1)}-${pad(now.getDate())} ${pad(now.getHours())}:${pad(now.getMinutes())}`; pocket.unshift({text: text.trim(), timestamp}); if (pocket.length > 50) pocket = pocket.slice(0, 50); // Batas maksimum 50 prompt localStorage.setItem('prompt_pocket', JSON.stringify(pocket)); renderPromptPocket(); } function loadPromptPocket() { try { return JSON.parse(localStorage.getItem('prompt_pocket')) || []; } catch { return []; } } function renderPromptPocket() { const listDiv = document.getElementById('promptPocketList'); const pocket = loadPromptPocket(); if (!listDiv) return; if (!pocket.length) { listDiv.innerHTML = `
Belum ada prompt yang disimpan.
Salin prompt apapun untuk menyimpannya di sini.
`; return; } listDiv.innerHTML = pocket.map((item, idx) => `
${item.text.replace(/
Disimpan: ${item.timestamp}
`).join(''); } function deletePromptFromPocket(idx) { let pocket = loadPromptPocket(); if (idx >= 0 && idx < pocket.length) { pocket.splice(idx, 1); localStorage.setItem('prompt_pocket', JSON.stringify(pocket)); renderPromptPocket(); } } // Render Prompt Pocket saat membuka layar Prompt Pocket document.querySelector('a.menu-item[href="#"][onclick*="prompt-pocket"]').addEventListener('click', function() { setTimeout(renderPromptPocket, 100); // delay agar layar sudah terlihat }); // Juga render saat halaman pertama kali dibuka window.addEventListener('DOMContentLoaded', renderPromptPocket); // ONBOARDING LOGIC function hideOnboarding() { const overlay = document.getElementById('onboardingOverlay'); overlay.classList.add('hide'); setTimeout(() => { overlay.style.display = 'none'; }, 350); // Optional: set flag in localStorage to not show onboarding again localStorage.setItem('onboarding_shown', '1'); } function onboardingLearnMore() { alert('Menampilkan tutorial atau informasi lebih lanjut'); } // Show onboarding only if not shown before window.addEventListener('DOMContentLoaded', function() { if (!localStorage.getItem('onboarding_shown')) { document.getElementById('onboardingOverlay').style.display = 'flex'; } else { document.getElementById('onboardingOverlay').style.display = 'none'; } }); } });
